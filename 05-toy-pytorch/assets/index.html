核心概念 — Value 类

class Value:
    def __init__(self, data, _children=(), _op=''):
        self.data = data          # 标量值
        self.grad = 0.0           # 梯度累积
        self._backward = lambda: None  # 局部梯度闭包
        self._prev = set(_children)   # 计算图子节点

    def __add__(self, other):
        out = Value(self.data + other.data, (self, other), '+')
        def _backward():
            self.grad += out.grad
            other.grad += out.grad
        out._backward = _backward
        return out
